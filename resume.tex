% size of the boxes used to align text
\newlength{\spacebox}
\settowidth{\spacebox}{123456789}
% vertical space separator between entries
\newcommand{\sepspace}{\vspace*{1em}}

%name macro
\newcommand{\name}[1]{
    \Huge%font size
    \fontfamily{ptm}\selectfont %font family
    %print name centered and bold
    \begin{center} \textbf{#1} \end{center}\par
    %back to normal size and font
    \normalsize
    \normalfont
}

%info macro
\newcommand{\info}[2]{
    %set indentation for personal information
    \noindent
    \hangindent=2em
    \hangafter=0
    %create a box to align two pieceso of text
    \parbox{\spacebox}{\textsl{#1}} \hspace{1em} #2 \par
} %entry value
