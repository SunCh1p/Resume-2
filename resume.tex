%name macro
\newcommand{\name}[1]{
    \Huge%font size
    \fontfamily{ptm}\selectfont %font family
    %print name centered and bold
    \begin{center} \textbf{#1} \end{center}\par
    %back to normal size and font
    \normalsize
    \normalfont
}

%info macro
\newcommand{\info}[2]{
    %set indentation for personal information
    \noindent
    \hangindent=2em
    \hangafter=0
    %create a box to align two pieceso of text
    \parbox{\spacebox}{\textsl{#1}} \hspace{1em} #2 \par
} %entry value



%Subsection dated macro
\newcommand{\datedsubsection}[2]{%
  \subsection*{\normalsize \normalfont #1 \hfill #2}%
}

\newcommand{\skill}[2]{%

    \begin{minipage}[t]{0.55\textwidth} % Right column (description)
        #2 % Multi-line description
    \end{minipage}%
    \vspace{0.8em}
    \par
}
