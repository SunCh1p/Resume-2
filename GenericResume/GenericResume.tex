\newcommand{\horizontalBox}[3]{
    \vspace{1em}
    \noindent \textsl{#1} \textbar{} \textsl{#2} \textbar{} \textsl{#3}
    \par
    \vspace{1em}
    \noindent\rule{\linewidth}{1pt}% draws a thin line

}

\newcommand{\splitbox}[2]{
    \par\vspace{1em}
    \noindent\makebox[\linewidth][l]{
        \parbox[t]{0.25\linewidth}{\large \textbf{#1} \normalsize}
        \parbox[t]{0.75\linewidth}{#2}%
    }
}

}
